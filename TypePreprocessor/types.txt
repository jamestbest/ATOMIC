TYPE INTEGER PREFIX i OVER 1 2 4 8
TYPE NATURAL PREFIX n OVER 1 2 4 8
TYPE REAL PREFIX f OVER 4 8 10
TYPE RATIONAL PREFIX q OVER 4 8 16
TYPE CHAR chr
TYPE STRING str
TYPE BOOLEAN bool
TYPE VOID void

ALIAS NUMERIC = INTEGER NATURAL REAL RATIONAL
ALIAS ALPHABETICAL = CHAR STRING

CHAR <-> INTEGER|NATURAL
INTEGER <-> NATURAL
NUMERIC|ALPHABETICAL|NOT_A_VALUE -> BOOLEAN

PLUS=NUMERIC; CHAR; POINTER || INTEGER|NATURAL;
MINUS=NUMERIC; ALPHABETICAL; POINTER && INTEGER|NATURAL;

MULT=NUMERIC; CHAR || INTEGER|NATURAL;
